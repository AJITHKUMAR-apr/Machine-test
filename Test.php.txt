<?php
function string_transform($inputStr,$position,$place){
    $outputStr = "invalid request";
    $strArray = str_split($inputStr);
    $element=$strArray[$position-1];
    if(($position+$place) < count($strArray)){
        $outArray = [];
        $i=0;
        foreach($strArray as $index => $charItem){
            if($charItem==$element){
              continue;  
            }
            if($index==($position+$place-1)){
                $outArray[$i]=$charItem;
                $outArray[$i+1]=$element;
            }else if($index>($position+$place-1)){
                $outArray[$i+1]=$charItem;
            }else{
                $outArray[$i]=$charItem;
            }
            $i++;
        }
        $outputStr=implode("",$outArray);
    }
    echo $outputStr;
}

$out=string_transform("Elephant",3,8);
echo $out;
